CREATE TABLE Account (
    Account_Id SERIAL PRIMARY KEY,
    Username VARCHAR(50) NOT NULL UNIQUE,
    Password_Hash VARCHAR(255) NOT NULL,            -- Mật khẩu mã hóa (BCrypt/Argon2)
    Email VARCHAR(100) UNIQUE,
    Display_Name VARCHAR(100),
    Avatar_Url TEXT,
    Status VARCHAR(20) DEFAULT 'offline',            -- Trạng thái (online, offline, busy, away...)
    Created_At TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE Conversation (
    Conversation_Id SERIAL PRIMARY KEY,
    Name VARCHAR(100),
    Type VARCHAR(20) NOT NULL DEFAULT 'private',    -- Loại: 'private' (1-1) hoặc 'group'
    Created_By INT NOT NULL,
    Created_At TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (Created_By) REFERENCES Account(Account_Id) ON DELETE CASCADE
);

CREATE TABLE Message (
    Message_Id SERIAL PRIMARY KEY,
    Conversation_Id INT NOT NULL,
    Sender_Id INT NOT NULL,
    Content TEXT,
    Message_Type VARCHAR(20) DEFAULT 'text',
    Created_At TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    Updated_At TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    Is_Read BOOLEAN DEFAULT FALSE,
    FOREIGN KEY (Conversation_Id) REFERENCES Conversation(Conversation_Id) ON DELETE CASCADE,
    FOREIGN KEY (Sender_Id) REFERENCES Account(Account_Id) ON DELETE CASCADE
);

CREATE TABLE Participants (
    Conversation_Id INT NOT NULL,
    Account_Id INT NOT NULL,
    Role VARCHAR(20) DEFAULT 'member',                -- member, admin, owner
    Joined_At TIMESTAMP DEFAULT CURRENT_TIMESTAMP,    -- Thời gian tham gia
    PRIMARY KEY (Conversation_Id, Account_Id),
    FOREIGN KEY (Conversation_Id) REFERENCES Conversation(Conversation_Id) ON DELETE CASCADE,
    FOREIGN KEY (Account_Id) REFERENCES Account(Account_Id) ON DELETE CASCADE
);

CREATE TABLE Attachment (
    Attachment_Id SERIAL PRIMARY KEY,                 -- ID file đính kèm
    Message_Id INT NOT NULL,                          -- Gắn vào tin nhắn nào
    File_Url TEXT NOT NULL,                           -- Link file (local hoặc cloud như S3)
    File_Type VARCHAR(50),                            -- Loại file: image/png, video/mp4...
    File_Size BIGINT,                                 -- Kích thước file (bytes)
    Uploaded_At TIMESTAMP DEFAULT CURRENT_TIMESTAMP,  -- Thời gian upload
    FOREIGN KEY (Message_Id) REFERENCES Message(Message_Id) ON DELETE CASCADE
);

CREATE TABLE Message_Status (
    Message_Id INT NOT NULL,
    Account_Id INT NOT NULL,
    Status VARCHAR(20) DEFAULT 'sent',                -- sent, delivered, read
    Updated_At TIMESTAMP DEFAULT CURRENT_TIMESTAMP,    -- Cập nhật trạng thái
    PRIMARY KEY (Message_Id, Account_Id),
    FOREIGN KEY (Message_Id) REFERENCES Message(Message_Id) ON DELETE CASCADE,
    FOREIGN KEY (Account_Id) REFERENCES Account(Account_Id) ON DELETE CASCADE
);

CREATE TABLE User_Contact (
    Contact_Id SERIAL PRIMARY KEY,
    Owner_Id INT NOT NULL,                            -- User sở hữu danh bạ
    Contact_User_Id INT NOT NULL,                     -- Người trong danh bạ
    Status VARCHAR(20) DEFAULT 'pending',             -- pending, accepted, blocked
    Created_At TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (Owner_Id) REFERENCES Account(Account_Id) ON DELETE CASCADE,
    FOREIGN KEY (Contact_User_Id) REFERENCES Account(Account_Id) ON DELETE CASCADE,
    CONSTRAINT Unique_Contact UNIQUE (Owner_Id, Contact_User_Id) -- tránh trùng
);

CREATE TABLE Notification (
    Notification_Id SERIAL PRIMARY KEY,
    Account_Id INT NOT NULL,                          -- Ai nhận thông báo
    Type VARCHAR(50) NOT NULL,                        -- friend_request, message, system...
    Content TEXT,                                     -- Nội dung thông báo
    Is_Read BOOLEAN DEFAULT FALSE,                    -- Đã đọc hay chưa
    Created_At TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (Account_Id) REFERENCES Account(Account_Id) ON DELETE CASCADE
);
CREATE TABLE verify_token (
    Id           INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    Email        VARCHAR(255) NOT NULL UNIQUE,
    Token        VARCHAR(255) NOT NULL UNIQUE,
    Expires_at   TIMESTAMP WITHOUT TIME ZONE NOT NULL,
    Password     VARCHAR(255) NOT NULL,
    Full_name    VARCHAR(255) NOT NULL
);
CREATE TABLE forgot_password (
    Id         SERIAL PRIMARY KEY,                       -- ID duy nhất
    Account_Id INT NOT NULL,                             -- Người dùng yêu cầu reset
    Token      VARCHAR(255) NOT NULL UNIQUE,             -- Mã reset password
    Expires_At TIMESTAMP WITHOUT TIME ZONE NOT NULL,     -- Hạn sử dụng token
    FOREIGN KEY (Account_Id) REFERENCES Account(Account_Id) ON DELETE CASCADE
);
